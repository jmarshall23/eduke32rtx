function(filter_existing out_var)
    set(result "")
    foreach(f IN LISTS ARGN)
        if(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${f}")
            list(APPEND result "${f}")
        endif()
    endforeach()
    set(${out_var} "${result}" PARENT_SCOPE)
endfunction()
